<!DOCTYPE html>
<html ng-app="myApp">

  <meta charset="UTF-8">
  <title>AngularJS Controller Demo</title>
  <script src="https://ajax.googleapis.com/ajax/libs/angularjs/1.8.2/angular.min.js"></script>
  </head>
<body ng-controller="UserController">

  <h2>User Information</h2>

  <form>
    <label>Name:</label>
    <input type="text" ng-model="user.Name" />

  </form>

  <div class="output">
    <h3>Live Preview:</h3>
    <p><strong>Hello,</strong> {{ user.Name }}</p>
  </div>

  <script>
    // Define AngularJS app and controller
    angular.module('myApp', [])
      .controller('UserController', function($scope) {
        $scope.user = {
          Name: '',
        };
      });
  </script>

</body>

<!DOCTYPE html>
<html ng-app="MyApp">
<head>
  <meta charset="UTF-8">
  <title>Html Form</title>
  <script src="https://ajax.googleapis.com/ajax/libs/angularjs/1.8.2/angular.min.js"></script>
  </head>
<body ng-controller="FormController">

  <h2>Raj Enterprise Inquiry Form</h2>

  <form>
    <label>First Name:</label>
    <input type="text" ng-model="formData.firstname" />

    <label>Last Name:</label>
    <input type="text" ng-model="formData.lastname" />

    <label>Contact Number:</label>
    <input type="text" ng-model="formData.contact" />

    <label>Date of Birth:</label>
    <input type="date" ng-model="formData.dob" />
  </form>
  <div class="output">
    <h3>Filtered Output:</h3>
    <p><strong>First Name (uppercase):</strong> {{ formData.firstname | uppercase }}</p>
    <p><strong>Last Name (lowercase):</strong> {{ formData.lastname | lowercase }}</p>
    <p><strong>Contact Number (number):</strong> {{ formData.contact | number }}</p>
    <p><strong>Date of Birth (formatted):</strong> {{ formData.dob | date:'fullDate' }}</p>
  </div>

  <script>
    angular.module('MyApp', [])
      .controller('FormController', function($scope) {
        $scope.formData = {
          firstname: '',
          lastname: '',
          contact: '',
          dob: null
        };
      });
  </script>

</body>
</html>
